How to use

Load order matters — PrimeVue Lara CSS first, our token sheet after, then per-component overrides via :pt.

// main.ts (Vue 3 + PrimeVue)
import { createApp } from 'vue'
import PrimeVue from 'primevue/config'
import 'primevue/resources/themes/lara-light-blue/theme.css'  // 1. Lara
import 'primevue/resources/primevue.min.css'
import 'primeicons/primeicons.css'
import './assets/pixarV2-tokens.css'                          // 2. our overrides

createApp(App).use(PrimeVue, { ripple: true }).mount('#app')
